\makeatletter
\newcommand{\newexample}[2]{%
	\newenvironment{#1}[1][]{%
		\par%
		\def \param{##1}%
		\normalfont \topsep6\p@\@plus6\p@\relax%
		\trivlist%
		\item \relax%
		{%
			\itshape%
			#2%
			\ifx\param\empty%
				\relax%
			\else%
				\ (\param)%
			\fi%
			\@addpunct{.}
		}%
		\hspace%
		\labelsep%
		\ignorespaces%
	}{%
		\endtrivlist\@endpefalse
	}
}
\makeatother

\newexample{example}{Exemple}
\newexample{solution}{Solution}
\newexample{question}{Question}
\newexample{answer}{Réponse}
